
POST /api/deliverables/{deliverableId}/submissions

sequenceDiagram
    actor Student as Student (Team Leader)
    participant SubmissionController
    participant SubmissionService
    participant GroupMembershipRepository
    participant DeliverableRepository
    participant CommitteeRepository
    participant SubmissionRepository
    participant NotificationService

    Student->>SubmissionController: POST /api/deliverables/{deliverableId}/submissions<br/>{ markdownContent }<br/>Authorization: Bearer <student_jwt>

    SubmissionController->>SubmissionController: extract requesterUUID from SecurityContext
    SubmissionController->>SubmissionService: createSubmission(deliverableId, groupId, requesterUUID, content)

    SubmissionService->>GroupMembershipRepository: findByGroupIdAndStudentId(groupId, requesterUUID)
    GroupMembershipRepository-->>SubmissionService: Optional<GroupMembership>

    alt not TEAM_LEADER
        SubmissionService-->>SubmissionController: throw ForbiddenException
        SubmissionController-->>Student: 403 {"error": "Only the Team Leader can submit deliverables"}
    end

    SubmissionService->>DeliverableRepository: findById(deliverableId)
    DeliverableRepository-->>SubmissionService: Deliverable

    alt currentDate > deliverable.submissionDeadline
        SubmissionService-->>SubmissionController: throw DeadlineExpiredException
        SubmissionController-->>Student: 400 {"error": "Submission rejected. The deadline has passed."}
    end

    SubmissionService->>CommitteeRepository: findByGroupId(groupId)
    CommitteeRepository-->>SubmissionService: Optional<Committee>

    alt group is NOT mapped to a committee
        SubmissionService-->>SubmissionController: throw MissingCommitteeException
        SubmissionController-->>Student: 400 {"error": "Group must be assigned to a committee before submitting"}
    end

    SubmissionService->>SubmissionRepository: save(DeliverableSubmission{content, submittedAt=now})
    SubmissionRepository-->>SubmissionService: savedSubmission

    Note over NotificationService: Trigger asynchronous notification to Advisor + Jury
    SubmissionService-)NotificationService: dispatchSubmissionNotification(committeeId, groupId, submissionId)

    SubmissionService-->>SubmissionController: SubmissionResponseDTO
    SubmissionController-->>Student: 201 { submissionId, submittedAt, status: "SUBMITTED" }

 GET /api/submissions/{submissionId}
 sequenceDiagram
    actor User as Student / Professor
    participant SubmissionController
    participant SubmissionService
    participant SubmissionRepository
    participant AccessControlService

    User->>SubmissionController: GET /api/submissions/{submissionId}<br/>Authorization: Bearer <jwt>

    SubmissionController->>SubmissionController: extract userUUID and role from SecurityContext
    SubmissionController->>SubmissionService: getSubmission(submissionId, userUUID, role)

    SubmissionService->>SubmissionRepository: findById(submissionId)
    SubmissionRepository-->>SubmissionService: DeliverableSubmission

    alt submission not found
        SubmissionService-->>SubmissionController: throw NotFoundException
        SubmissionController-->>User: 404 {"error": "Submission not found"}
    end

    SubmissionService->>AccessControlService: verifyReadAccess(submission, userUUID, role)
    AccessControlService-->>SubmissionService: boolean (isAuthorized)

    alt user is Student NOT in the group OR Professor NOT in the committee
        SubmissionService-->>SubmissionController: throw ForbiddenException
        SubmissionController-->>User: 403 {"error": "You do not have permission to view this submission"}
    end

    SubmissionService-->>SubmissionController: SubmissionDetailDTO
    SubmissionController-->>User: 200 { submissionId, content, submittedAt, ... }
